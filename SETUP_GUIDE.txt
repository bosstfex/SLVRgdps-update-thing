💻 CODE EXAMPLE - Using the Menus:
──────────────────────────────────────────────────────────────────────────────

// In your game code, you can now use:

// Open search menu
window.menuSystem.openSearch((levelId) => {
  console.log('Loading level:', levelId);
});

// Open level select
window.menuSystem.openLevelSelect([
  { id: '1', name: 'Stereo Madness' },
  { id: '2', name: 'Back on Track' }
], (level) => {
  console.log('Selected:', level);
});

// Open featured levels
window.menuSystem.openFeatured([
  { name: 'Challenge 1', description: 'Hard level', id: '100' }
], (level) => {
  console.log('Playing:', level);
});

// Close any open menu
window.menuSystem.hide();

// Check status
if (window.menuSystem.isMenuOpen()) {
  console.log('Current menu:', window.menuSystem.getCurrentMenu());
}
