🖥️  HOW TO USE IN CODE:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// Open search menu
window.menuSystem.openSearch((levelId) => {
  console.log('Load level:', levelId);
});

// Open level select
window.menuSystem.openLevelSelect([
  { id: '1', name: 'Level 1' }
], (level) => {
  console.log('Play:', level);
});

// Open featured
window.menuSystem.openFeatured([
  { name: 'Cool', description: 'Nice level' }
], (level) => {
  console.log('Play:', level);
});

// Close menu
window.menuSystem.hide();
