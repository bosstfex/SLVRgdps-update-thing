💻 HOW IT WORKS:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. index.html loads menu-system.css and menu-system.js in the <head>
2. index.html fetches menu-system.html content and injects it into the body
3. menu-system.js initializes automatically when DOM is ready
4. Game code calls window.menuSystem functions to open/close menus
5. All menus stay hidden until explicitly opened


🚀 QUICK START (After files created):
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// Open search menu
window.menuSystem.openSearch((levelId) => {
  console.log('Loading level:', levelId);
});

// Open level select
window.menuSystem.openLevelSelect([
  { id: '1', name: 'Level 1' }
], (level) => {
  console.log('Selected:', level);
});

// Open featured
window.menuSystem.openFeatured([
  { name: 'Featured', description: 'Cool level', id: '100' }
], (level) => {
  console.log('Playing:', level);
});

// Close any menu
window.menuSystem.hide();
